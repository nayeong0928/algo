import java.util.Scanner;
import java.util.Stack;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int[] arr = new int[N];
        for (int i = 0; i < N; i++) {
            arr[i] = sc.nextInt();
        }
        sc.close();
        System.out.println(stack(arr));
    }

    private static String stack(int[] arr) {

        Stack<Integer> stack = new Stack<>();
        StringBuffer sb = new StringBuffer();
        int num = 1;

        for (int now = 0; now < arr.length; now++) {
            if (num <= arr[now]) {
                while (num <= arr[now]) {
                    stack.push(num++);
                    sb.append("+\n");
                }
                stack.pop();
                sb.append("-\n");
            } else {
                if (stack.pop() > arr[now]) {
                    return "NO";
                } else {
                    sb.append("-\n");
                }
            }
        }

        return sb.toString();
    }
}
