# ABCDE

## 문제 분석

- A->B, B->C, C->D, D->E 인 친구 관계가 있는지 찾는 문제
- DFS로 A를 방문하여 깊이가 5까지 되면 true

## 구현

```java

import java.util.ArrayList;
import java.util.Scanner;

public class sol13023 {

    static ArrayList<Integer>[] arr;
    static int N, M;
    static boolean[] visited;
    static boolean arrive;

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        N = sc.nextInt();
        M = sc.nextInt();
        arr = new ArrayList[N];
        visited = new boolean[N];
        arrive = false;

        for (int i = 0; i < N; i++) {
            arr[i] = new ArrayList<>();
        }

        for (int i = 0; i < M; i++) {
            int a = sc.nextInt();
            int b = sc.nextInt();
            arr[a].add(b);
            arr[b].add(a);
        }

        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                DFS(i, 1);
            }

            if (arrive) {
                break;
            }
        }

        System.out.println(arrive ? "1" : "0");

    }

    private static void DFS(int n, int depth) {

        if (depth == 5) {
            arrive = true;
            return;
        }

        visited[n] = true;

        for (int v : arr[n]) {
            if (!visited[v]) {
                DFS(v, depth + 1);
            }
        }

        visited[n] = false;
    }
}

```
